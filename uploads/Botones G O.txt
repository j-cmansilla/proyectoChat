/*Botón grabar*/
if(lblPath.getText()!="")
        {
            String strError="";
            if(LlenarArchivo(lblPath.getText(), txtIngreso.getText(), strError))
            {
                JOptionPane.showMessageDialog(null, "Se ingreso correctamente el registro: " +
                        txtIngreso.getText(), "Guardar", WIDTH);
                txtIngreso.setText("");
            }
            else
            {
                JOptionPane.showMessageDialog(null, "Se produjo el siguiente error al guardar: " +
                        strError,"Error", WIDTH);
            }
        }
        else
        {
            JOptionPane.showMessageDialog(null, "Debe seleccionar un archivo para guardar los registros ",
                    "ERROR!", WIDTH);
        
}


/*Botón Obtener*/
if(lblPath.getText()!="")
        {
            String strError="";
            if(!Obtener(lblPath.getText(), strError))
            {
                JOptionPane.showMessageDialog(null, "Se prodrujo un error al cagar los registros: " +strError,"ERROR!", WIDTH);
            }
            
        }
        else
        {
            JOptionPane.showMessageDialog(null, "Debe seleccionar un archivo para obtener los registros ","ERROR!", WIDTH);
        }
